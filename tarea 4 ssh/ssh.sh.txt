#!/bin/bash

configurar_ssh() {

    echo "===== CONFIGURAR SSH LINUX ====="

    #instalar OpenSSH si no esta
    if dpkg -l | grep -q openssh-server; then
        echo "SSH ya esta instalado"
    else
        echo "Instalando SSH..."
        sudo apt-get update -y
        sudo apt-get install openssh-server -y
    fi

    # Iniciar el Servicio
    systemctl start ssh

    #Habilitar en arranque
    systemctl enable ssh

    #configurar firewall (UFW)
    if command -v ufw >/dev/null 2>&1; then
        ufw allow 22/tcp
        ufw --force enable
    else
        echo "UFW no instalado, omitiendo firewall"
    fi

    echo "SSH configurado correctamente"
}
