#!/bin/bash

echo "===== INSTALACION DHCP LINUX ====="

if dpkg -l | grep -q isc-dhcp-server; then
    echo "DHCP ya esta instalado"
else
    echo "Instalando DHCP..."
    sudo apt-get update
    sudo apt-get install isc-dhcp-server -y
fi

echo "===== CONFIGURACION DHCP ====="

read -p "nombre del Scope: " scope
read -p "IP inicial: " ip_inicio
read -p "IP final: " ip_fin
read -p "Gateway: " gateway
read -p "DNS: " dns
read -p "Tiempo de concesion (segundos): " lease

sudo bash -c "cat > /etc/dhcp/dhcpd.conf" <<EOL
default-lease-time $lease;
max-lease-time $lease;

subnet 192.168.100.0 netmask 255.255.255.0 {
    range $ip_inicio $ip_fin;
    option routers $gateway;
    option domain-name-servers $dns;
}
EOL
